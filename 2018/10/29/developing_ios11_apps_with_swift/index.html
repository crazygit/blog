import xml.etree.ElementTree as ET
import requests
root = ET.fromstring(requests.get('http://podcasts.apple.com/stanford/developing_ios11_apps.xml').text)
for enclosure in root.findall('.//enclosure'):
    print(enclosure.get('url'))
